\draw[dashed,red,thick] (-3,-3) -| ++(1,2) -| ++(1,2) -| ++(1,2) -| ++(1,2) -| ++(1,2);
% draw line and point

	\draw [thick] (\x cm,3pt) -- (\x cm,-3pt) node[below] {$\x$};

	\draw [thick] (3pt,\y cm) -- (-3pt,\y cm) node[left] {$\y$};
